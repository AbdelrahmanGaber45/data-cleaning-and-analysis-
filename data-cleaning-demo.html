04 — Python Script Used
clean_sales_data.py
import pandas as pd

# Load raw data
df = pd.read_excel("sales_raw.xlsx")

# 1. Standardize city names
city_map = {
    "cairo": "Cairo", "CAIRO": "Cairo", "القاهرة": "Cairo",
    "ALEXANDRIA": "Alexandria", "alex": "Alexandria",
    "الجيزة": "Giza", "giza": "Giza"
}
df["City"] = df["City"].map(city_map).fillna(df["City"])

# 2. Remove duplicate transactions
df = df.drop_duplicates(subset=["Transaction_ID"], keep="first")

# 3. Fill missing sales values
df["Sales_Amount"] = df.groupby("City")["Sales_Amount"].transform(
    lambda x: x.fillna(x.mean())
)

# 4. Normalize date format
df["Date"] = pd.to_datetime(df["Date"], dayfirst=True).dt.strftime("%Y-%m-%d")

# Export cleaned file
df.to_excel("sales_cleaned.xlsx", index=False)
print(f"Done! {len(df)} clean records saved.")
